<PythonRunner code={`
import numpy as np
import matplotlib.pyplot as plt

# Generate data
a = np.random.randn(1000)
b = np.random.randn(1000)
X = np.column_stack((np.ones_like(a), a))
y = 2 * a + b  # Linear relation: y = 2a + 3

# Least squares
beta, _, _, _ = np.linalg.lstsq(X, y, rcond=None)
y_pred = X @ beta

# Sort for smooth line
sorted_idx = np.argsort(a)
a_sorted = a[sorted_idx]
y_pred_sorted = y_pred[sorted_idx]

# Plot
plt.scatter(a, y, label='Data', alpha=0.5)
plt.plot(a_sorted, y_pred_sorted, color='red', label='Fitted Line')
plt.xlabel('a')
plt.ylabel('y')
plt.title('Least Squares Fit (1 Feature)')
plt.legend()
plt.show()`
